283  bool create(uint32_t depth, uint32_t csp, uint32_t numInstances)
284  {
285  uint32_t numPartition = MAX_NUM_PARTITIONS >> (depth * 2);
286  uint32_t cuSize = g_maxCUSize >> depth;
287  uint32_t sizeL = cuSize * cuSize;
288  uint32_t sizeC = sizeL >> (CHROMA_H_SHIFT(csp) + CHROMA_V_SHIFT(csp));
289  CHECKED_MALLOC(trCoeffMemBlock, coeff_t, (sizeL + sizeC * 2) * numInstances);
290  CHECKED_MALLOC(charMemBlock, uint8_t, numPartition * numInstances * CUData::BytesPerPartition);
291  CHECKED_MALLOC(mvMemBlock, MV, numPartition * 4 * numInstances);
292  return true;
293 
294  fail:
295  return false;
296  }
297 
298  void destroy()
299  {
300  X265_FREE(trCoeffMemBlock);
301  X265_FREE(mvMemBlock);
302  X265_FREE(charMemBlock);
